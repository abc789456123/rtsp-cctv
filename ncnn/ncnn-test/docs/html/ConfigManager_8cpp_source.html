124std::string ConfigManager::parseJsonString(const std::string& json, const std::string& key) {
125 std::string searchKey = "\"" + key + "\"";
126 size_t pos = json.find(searchKey);
127 if (pos == std::string::npos) return "";
128
129 pos = json.find(":", pos);
130 if (pos == std::string::npos) return "";
131
132 pos = json.find("\"", pos);
133 if (pos == std::string::npos) return "";
134 pos++; // Skip opening quote
135
136 size_t endPos = json.find("\"", pos);
137 if (endPos == std::string::npos) return "";
138
139 return json.substr(pos, endPos - pos);
140}

147 pos = json.find(":", pos);
148 if (pos == std::string::npos) return defaultValue;
149 pos++; // Skip colon

190bool ConfigManager::parseJsonBool(const std::string& json, const std::string& key, bool defaultValue) {
191 std::string searchKey = "\"" + key + "\"";
192 size_t pos = json.find(searchKey);
193 if (pos == std::string::npos) return defaultValue;
194
195 pos = json.find(":", pos);
196 if (pos == std::string::npos) return defaultValue;
197
198 size_t truePos = json.find("true", pos);
199 size_t falsePos = json.find("false", pos);
200
201 if (truePos != std::string::npos && (falsePos == std::string::npos || truePos < falsePos)) {
202 return true;
203 } else if (falsePos != std::string::npos) {
204 return false;
205 }
206
207 return defaultValue;
208}
