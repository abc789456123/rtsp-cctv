124void MetadataPublisher::publishingLoop() {
125 while (running_) {
126 DetectionMetadata metadata;
127 bool has_metadata = false;
128
129 // Get metadata from queue
130 {
131 std::lock_guard<std::mutex> lock(queue_mutex_);
132 if (!metadata_queue_.empty()) {
133 metadata = metadata_queue_.front();
134 metadata_queue_.pop();
135 has_metadata = true;
136 }
137 }
138
139 if (has_metadata) {
140 // Create JSON and send
141 std::string json_data = createJsonMetadata(metadata);
142
143 // For debugging, print to console instead of HTTP POST
144 std::cout << "=== Metadata JSON ===" << std::endl;
145 std::cout << json_data << std::endl;
146 std::cout << "===================" << std::endl;
147
148 // Attempt HTTP POST (will fail if no server, but that's OK for demo)
149 if (sendHttpPost(json_data)) {
150 published_count_++;
151 }
152 }
153
154 std::this_thread::sleep_for(std::chrono::milliseconds(publish_interval_ms_));
155 }
156}

196bool MetadataPublisher::sendHttpPost(const std::string& json_data) {
197 CURL *curl;
198 CURLcode res;
199 std::string response_data;
200 bool success = false;
201
202 curl = curl_easy_init();
203 if (curl) {
204 std::string url = "http://" + host_ + ":" + std::to_string(port_) + endpoint_;
205
206 curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
207 curl_easy_setopt(curl, CURLOPT_POSTFIELDS, json_data.c_str());
208 curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, WriteCallback);
209 curl_easy_setopt(curl, CURLOPT_WRITEDATA, &response_data);
210 curl_easy_setopt(curl, CURLOPT_TIMEOUT, 5L); // 5 second timeout
211
212 // Set content type header
213 struct curl_slist *headers = NULL;
214 headers = curl_slist_append(headers, "Content-Type: application/json");
215 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);
216
217 res = curl_easy_perform(curl);
218
219 if (res == CURLE_OK) {
220 long response_code;
221 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &response_code);
222 if (response_code == 200) {
223 success = true;
224 } else {
225 std::cout << "HTTP POST failed with response code: " << response_code << std::endl;
226 }
227 } else {
228 // Don't spam error messages - server might not be running
229 // std::cerr << "HTTP POST failed: " << curl_easy_strerror(res) << std::endl;
230 }
231
232 curl_slist_free_all(headers);
233 curl_easy_cleanup(curl);
234 }
235
236 return success;
237}

239std::string MetadataPublisher::formatTimestamp(const std::chrono::system_clock::time_point& timestamp) {
240 auto time_t = std::chrono::system_clock::to_time_t(timestamp);
241 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(timestamp.time_since_epoch()) % 1000;
242
243 std::stringstream ss;
244 ss << std::put_time(std::gmtime(&time_t), "%Y-%m-%dT%H:%M:%S");
245 ss << "." << std::setfill('0') << std::setw(3) << ms.count() << "Z";
246
247 return ss.str();
248}
