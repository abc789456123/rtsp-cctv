86int YoloDetector::detect(const cv::Mat& bgr, std::vector<Object>& objects, float prob_threshold, float nms_threshold)
87{
88 int img_w = bgr.cols;
89 int img_h = bgr.rows;
90
91 // letterbox pad to multiple of 32
92 int w = img_w;
93 int h = img_h;
94 float scale = 1.f;
95 if (w > h)
96 {
97 scale = (float)target_size / w;
98 w = target_size;
99 h = h * scale;
100 }
101 else
102 {
103 scale = (float)target_size / h;
104 h = target_size;
105 w = w * scale;
106 }
107
108 ncnn::Mat in = ncnn::Mat::from_pixels_resize(bgr.data, ncnn::Mat::PIXEL_BGR2RGB, img_w, img_h, w, h);
109
110 // pad to target_size rectangle
111 int wpad = (w + 31) / 32 * 32 - w;
112 int hpad = (h + 31) / 32 * 32 - h;
113 ncnn::Mat in_pad;
114 ncnn::copy_make_border(in, in_pad, hpad / 2, hpad - hpad / 2, wpad / 2, wpad - wpad / 2, ncnn::BORDER_CONSTANT, 114.f);
115
116 in_pad.substract_mean_normalize(mean_vals, norm_vals);
117
118 ncnn::Extractor ex = yolov4.create_extractor();
119 ex.input("data", in_pad);
120
121 ncnn::Mat out;
122 ex.extract("output", out);
123
124 objects.clear();
125
126 if (out.h > 0) {
127 for (int i = 0; i < out.h; i++) {
128 const float* detection = out.row(i);
129
130 int class_id = (int)detection[0];
131 float confidence = detection[1];
132
133 if (confidence < prob_threshold) continue;
134
135 // Convert normalized coordinates (0-1) to pixel coordinates
136 float x1 = detection[2] * img_w;
137 float y1 = detection[3] * img_h;
138 float x2 = detection[4] * img_w;
139 float y2 = detection[5] * img_h;
140
141 // Clamp to image bounds
142 x1 = std::max(std::min(x1, (float)(img_w - 1)), 0.f);
143 y1 = std::max(std::min(y1, (float)(img_h - 1)), 0.f);
144 x2 = std::max(std::min(x2, (float)(img_w - 1)), 0.f);
145 y2 = std::max(std::min(y2, (float)(img_h - 1)), 0.f);
146
147 // Skip invalid boxes
148 if (x2 <= x1 || y2 <= y1) continue;
149 if ((x2 - x1) < 10 || (y2 - y1) < 10) continue;
150
151 Object obj;
152 obj.label = class_id;
153 obj.prob = confidence;
154 obj.rect = cv::Rect_<float>(x1, y1, x2 - x1, y2 - y1);
155
156 objects.push_back(obj);
157 }
158 }
159
160 return 0;
161}

222void YoloDetector::qsort_descent_inplace(std::vector<Object>& faceobjects, int left, int right)
223{
224 int i = left;
225 int j = right;
226 float p = faceobjects[(left + right) / 2].prob;
227
228 while (i <= j)
229 {
230 while (faceobjects[i].prob > p)
231 i++;
232
233 while (faceobjects[j].prob < p)
234 j--;
235
236 if (i <= j)
237 {
238 std::swap(faceobjects[i], faceobjects[j]);
239 i++;
240 j--;
241 }
242 }
243
244 if (left < j)
245 qsort_descent_inplace(faceobjects, left, j);
246 if (i < right)
247 qsort_descent_inplace(faceobjects, i, right);
248}

257void YoloDetector::nms_sorted_bboxes(const std::vector<Object>& faceobjects, std::vector<int>& picked, float nms_threshold)
258{
259 picked.clear();
260
261 const int n = faceobjects.size();
262
263 std::vector<float> areas(n);
264 for (int i = 0; i < n; i++)
265 {
266 areas[i] = faceobjects[i].rect.area();
267 }
268
269 for (int i = 0; i < n; i++)
270 {
271 const Object& a = faceobjects[i];
272
273 int keep = 1;
274 for (int j = 0; j < (int)picked.size(); j++)
275 {
276 const Object& b = faceobjects[picked[j]];
277
278 float inter_area = intersection_area(a, b);
279 float union_area = areas[i] + areas[picked[j]] - inter_area;
280
281 if (inter_area / union_area > nms_threshold)
282 keep = 0;
283 }
284
285 if (keep)
286 picked.push_back(i);
287 }
