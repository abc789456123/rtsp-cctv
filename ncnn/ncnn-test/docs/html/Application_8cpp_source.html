170void Application::processFrame(const cv::Mat& frame) {
171 const auto& config = config_manager_->getConfig();
172
173 // Detect objects
174 std::vector<Object> objects;
175 yolo_detector_->detect(frame, objects, config.detection_threshold, config.nms_threshold);
176
177 if (!objects.empty()) {
178 detection_count_ += objects.size();
179 }
180
181 // Send frame to RTSP stream
182 cv::Mat display_frame = frame.clone();
183 if (config.draw_detections && !objects.empty()) {
184 YoloDetector::draw_objects(display_frame, objects);
185 }
186
187 rtsp_streamer_->pushFrame(display_frame);
188
189 // Publish metadata at configured interval
190 auto now = std::chrono::steady_clock::now();
191 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - last_metadata_time_);
192
193 if (elapsed.count() >= config.metadata_publish_interval_ms) {
194 metadata_publisher_->publishDetections(objects, frame.cols, frame.rows, "camera_" + std::to_string(config.camera_id));
195 last_metadata_time_ = now;
196 }
197
198 // Show display if enabled
199 if (config.show_display) {
200 // Add status overlay
201 std::string status = "Frame: " + std::to_string(frame_count_) +
202 " | Detections: " + std::to_string(objects.size()) +
203 " | Queue: " + std::to_string(metadata_publisher_->getQueueSize());
204
205 cv::putText(display_frame, status, cv::Point(10, 30),
206 cv::FONT_HERSHEY_SIMPLEX, 0.7, cv::Scalar(0, 255, 0), 2);
207
208 cv::imshow("AI Detection System", display_frame);
209 }
210}
